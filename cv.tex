    \section{Research Experience}

\textbf{PPRBench: A Process-level Benchmark for LLMs’ Physical
Reasoning} 
\vspace{0.05 cm}
\begin{onecolentry}
University of Science and Technology of China, Lab for Data Science
\begin{highlights}
    \item Designed PPRBench, a contamination-free benchmark for evaluating physical reasoning in large language models.
    \item Developed PhysGrader, a rubric-guided LLM-as-a-judge framework for process-level reasoning evaluation.
    \item Explored GRPO on PPRBench, with process reward generated by PhysGrader.
    \item Evaluated 16 frontier LLMs and analyzed reasoning fidelity in physics tasks.
    \item Paper submitted to SIGIR 2026 (Resource Track).
\end{highlights}
\end{onecolentry}

        \vspace{0.2cm}
    
            \textbf{An Integrated Simulated Annealing Algorithm for WiFi-based Indoor Positioning}
        \vspace{0.10 cm}
        \begin{onecolentry}
            \begin{highlights}
    \item Proposed a simulated-annealing-based optimization framework for Time-of-Flight (ToF) indoor localization using Channel State Information (CSI).
    \item Combined triangulation and the MUSIC algorithm for transmitter localization and developed a two-stage error mitigation pipeline including data cleaning and nonlinear optimization.
    \item Demonstrated improved positioning accuracy and robustness compared with traditional optimization methods such as L-BFGS.
            \end{highlights}
        \end{onecolentry}

        \vspace{0.2cm}
        
            \textbf{Fingerprint Reasoner: Theoretical Analysis of Raman Spectroscopy and Knowledge-Augmented Reasoning for Organic Compound Identification with Large Language Models}
        \vspace{0.10 cm}
        \begin{onecolentry}
            \begin{highlights}
                \item February 2025 - May 2025: Participated in the Quantum Optics and Nonlinear Optics Laboratory led by Doctoral Supervisor Zhiyuan Zhou, part of Academician Guangcan Guo's team at USTC, exploring areas related to Raman spectroscopy and AI for Science.
%                 \item Key contributions:
% • We identify two key limitations of solving spectroscopy problems with LLMs.
\item We propose a method called Fingerprint Reasoner, which combines
LLMs with knowledge retrieval and CoT.
\item  We conduct extensive experiments on QM9S dataset, demonstrating that
our method achieves high accuracy.
\item We provide a theoretical analysis of Raman spectroscopy, and reproduced
optical apparatus to obtain spectral data as well as extracted the features of
different categories of data, which supports our design of the method.
                \item This work provides a new perspective for the application of LLM in the field of spectroscopy, and also enlightens the prospect for on-site and real-time identification of organic compounds.

            \end{highlights}
        \end{onecolentry}
